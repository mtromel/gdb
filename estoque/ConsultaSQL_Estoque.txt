SELECT I.ITEMID
      ,IT.ITEMNAME
      ,D.INVENTLOCATIONID
      ,D.WMSLOCATIONID
      ,I.PHYSICALINVENT
      ,P.UNITID
      ,P.PRICE
      ,I.PHYSICALINVENT * P.PRICE AS TOTAL
  FROM [DynamicsAxProd].[dbo].[INVENTSUM] AS I
  LEFT JOIN [DynamicsAxProd].[dbo].INVENTDIM AS D ON I.INVENTDIMID = D.INVENTDIMID
  LEFT JOIN DynamicsAxProd.dbo.INVENTTABLE AS IT ON I.ITEMID = IT.ITEMID
  LEFT JOIN DynamicsAxProd.dbo.INVENTITEMPRICE AS P ON I.ITEMID = P.ITEMID
  WHERE I.PHYSICALINVENT > 0 AND I.DATAAREAID = 'gsb' AND D.DATAAREAID = 'gsb' AND P.VERSIONID = ' ' AND P.PRICETYPE = '0' AND P.DATAAREAID = 'gsb'
